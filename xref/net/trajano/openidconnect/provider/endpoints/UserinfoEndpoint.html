6   import javax.ws.rs.POST;
7   import javax.ws.rs.Path;

9   import javax.ws.rs.core.Context;
10  import javax.ws.rs.core.MediaType;
11  import javax.ws.rs.core.Response;

13  import net.trajano.openidconnect.provider.internal.AuthorizationUtil;
14  import net.trajano.openidconnect.provider.spi.KeyProvider;
15  import net.trajano.openidconnect.provider.spi.TokenProvider;
16  import net.trajano.openidconnect.provider.spi.UserinfoProvider;
17  import net.trajano.openidconnect.token.IdToken;
18  
19  /**
20   * <p>
21   * The UserInfo Endpoint is an OAuth 2.0 Protected Resource that returns Claims
22   * about the authenticated End-User. To obtain the requested Claims about the
23   * End-User, the Client makes a request to the UserInfo Endpoint using an Access
24   * Token obtained through OpenID Connect Authentication. These Claims are
25   * normally represented by a JSON object that contains a collection of name and
26   * value pairs for the Claims.
27   * </p>
28   * <p>
29   * Communication with the UserInfo Endpoint MUST utilize TLS. See Section 16.17
30   * for more information on using TLS.
31   * </p>
32   * <p>
33   * The UserInfo Endpoint MUST support the use of the HTTP GET and HTTP POST
34   * methods defined in RFC 2616 [RFC2616].
35   * </p>
36   * <p>
37   * The UserInfo Endpoint MUST accept Access Tokens as OAuth 2.0 Bearer Token
38   * Usage [RFC6750].
39   * </p>
40   * <p>
41   * The UserInfo Endpoint SHOULD support the use of Cross Origin Resource Sharing
42   * (CORS) [CORS] and or other methods as appropriate to enable Java Script
43   * Clients to access the endpoint.
44   * </p>
45   */
46  @Path("profile")
47  @Produces(MediaType.APPLICATION_JSON)
48  public class UserinfoEndpoint {
49  
50      @EJB
51      private TokenProvider tokenProvider;
52  
53      @EJB
54      private KeyProvider keyProvider;
55  
56      private UserinfoProvider userinfoProvider;
57  
58      /**
59       * <p>
60       * The Client sends the UserInfo Request using either HTTP GET or HTTP POST.
61       * The Access Token obtained from an OpenID Connect Authentication Request
62       * MUST be sent as a Bearer Token, per Section 2 of OAuth 2.0 Bearer Token
63       * Usage [RFC6750].
64       * </p>
65       * <p>
66       * It is RECOMMENDED that the request use the HTTP GET method and the Access
67       * Token be sent using the Authorization header field.
68       * </p>
69       *
70       * @param req
71       * @return
72       */
73      @GET
74      public Response getOp(@Context final HttpServletRequest req) {
75  
76          return op(req);
77      }
78  
79      @POST
80      public Response op(@Context final HttpServletRequest req) {
81  
82          final String accessToken = AuthorizationUtil.processBearer(req);
83          IdToken idToken = tokenProvider.getByAccessToken(accessToken)
84                  .getIdToken(keyProvider.getPrivateJwks());
85          return Response.ok(userinfoProvider.getUserinfo(idToken))
86                  .build();
87      }
88  
89      @EJB
90      public void setUserinfoProvider(final UserinfoProvider userinfoProvider) {
91  
92          this.userinfoProvider = userinfoProvider;
93      }
94  
95  }
